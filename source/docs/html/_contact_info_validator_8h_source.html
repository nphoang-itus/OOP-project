162 static void validateEmail(const std::string& email, ValidationResult& result) {
163 if (email.empty()) {
164 ContactInfoErrorHelper::addError(result, ContactInfoError::EMPTY_EMAIL);
165 return;
166 }
167 if (email.length() > MAX_EMAIL_LENGTH) {
168 ContactInfoErrorHelper::addError(result, ContactInfoError::EMAIL_TOO_LONG);
169 return;
170 }
171 if (!std::regex_match(email, EMAIL_REGEX)) {
172 ContactInfoErrorHelper::addError(result, ContactInfoError::INVALID_EMAIL_FORMAT);
173 }
174 }
175
181 static void validatePhone(const std::string& phone, ValidationResult& result) {
182 if (phone.empty()) {
183 ContactInfoErrorHelper::addError(result, ContactInfoError::EMPTY_PHONE);
184 return;
185 }
186 // Remove '+' prefix for length check
187 std::string phoneDigits = phone;
188 if (!phoneDigits.empty() && phoneDigits[0] == '+') {
189 phoneDigits = phoneDigits.substr(1);
190 }
191 if (phoneDigits.length() > MAX_PHONE_LENGTH) {
192 ContactInfoErrorHelper::addError(result, ContactInfoError::PHONE_TOO_LONG);
193 return;
194 }
195 if (!std::regex_match(phone, PHONE_REGEX)) {
196 ContactInfoErrorHelper::addError(result, ContactInfoError::INVALID_PHONE_FORMAT);
197 }
198 }
199
205 static void validateAddress(const std::string& address, ValidationResult& result) {
206 if (!address.empty()) {
207 if (address.length() > MAX_ADDRESS_LENGTH) {
208 ContactInfoErrorHelper::addError(result, ContactInfoError::ADDRESS_TOO_LONG);
209 return;
210 }
211 if (!std::regex_match(address, ADDRESS_REGEX)) {
212 ContactInfoErrorHelper::addError(result, ContactInfoError::INVALID_ADDRESS_FORMAT);
213 }
214 }
215 }
