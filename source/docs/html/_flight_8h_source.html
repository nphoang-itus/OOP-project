54 static std::vector<SeatNumber> generateSeatNumbers(const std::string &classCode, int count, const SeatClassMap &seatLayout)
55 {
56 std::vector<SeatNumber> seats;
57 seats.reserve(count);
58
59 // Determine padding based on count
60 int padding = count > 99 ? 3 : 2;
61
62 for (int i = 1; i <= count; i++)
63 {
64 std::stringstream ss;
65 ss << classCode << std::setfill('0') << std::setw(padding) << i;
66 auto seatNumberResult = SeatNumber::create(ss.str(), seatLayout);
67 if (seatNumberResult)
68 {
69 seats.push_back(*seatNumberResult);
70 }
71 }
72 return seats;
73 }

320 bool isSeatAvailable(const std::string &seatNumberStr) const
