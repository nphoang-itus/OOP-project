55 template<typename InputType>
56 static Result<Price> createInternal(const InputType& input) {
57 auto validationResult = PriceValidator::validate(input);
58 if (!validationResult.isValid()) {
59 return getValidationFailure<Price>(validationResult);
60 }
61
62 if constexpr (std::is_same_v<InputType, std::pair<double, std::string>>) {
63 const auto& [amount, currency] = input;
64 return Success(Price(amount, currency));
65 } else if constexpr (std::is_same_v<InputType, std::string>) {
66 auto parsed = PriceParser::parse(input);
67 if (!parsed.has_value()) {
68 return Failure<Price>(CoreError("Failed to parse price", "PARSE_ERROR"));
69 }
70 const auto& [amount, currency] = parsed.value();
71 return Success(Price(amount, currency));
72 }
73 return Failure<Price>(CoreError("Unsupported input type", "UNSUPPORTED_TYPE"));
74 }
