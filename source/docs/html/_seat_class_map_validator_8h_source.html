143 static void validateSpecific(const std::string& value, ValidationResult& result) {
144 // Validate format
145 std::regex pattern("^([A-Za-z0-9]+:[-]?[0-9]+)(,[A-Za-z0-9]+:[-]?[0-9]+)*$");

205 static void validateClassEntry(const std::string& entry, ValidationResult& result) {
206 size_t colonPos = entry.find(':');
207 if (colonPos != std::string::npos) {

211 // Validate case mixing
212 if (!isAllUppercase(className) && !isAllLowercase(className)) {

225 }
226
227 // Validate seat count
